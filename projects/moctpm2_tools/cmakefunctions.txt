cmake_minimum_required (VERSION 3.5)

function(buildexecutable libType targetName targetDir mssSrcDir buildDir apptype mssBinDir  ${libraryLists})
  # adding dsf include directories listed in dsf_includes.txt
  file(STRINGS "${targetDir}/mss_includes.txt" DSF_INCLUDES)
  string(REGEX REPLACE "MSS_SRC_DIR/" "${mssSrcDir}/" DSF_INCLUDES "${DSF_INCLUDES}")

  include_directories("." ${DSF_INCLUDES})

  message("${targetName} includes")
  message("----------------")
  foreach(dir ${DSF_INCLUDES})
      message(${dir})
  endforeach()

  file(STRINGS "${targetDir}/mss_sources.txt" DSF_SOURCES)
  if(WIN32)
    if (EXISTS "${targetDir}/mss_win_sources.txt")
      file(STRINGS "${targetDir}/mss_win_sources.txt" DSF_WIN_SOURCES)
      set(DSF_SOURCES ${DSF_SOURCES} ${DSF_WIN_SOURCES})
    endif()
  endif()
  string(REGEX REPLACE "MSS_SRC_DIR/" "${mssSrcDir}/" DSF_SOURCES "${DSF_SOURCES}")
  message("dsf sources")
  message("-----------")
  foreach(file ${DSF_SOURCES})
      message(${file})
  endforeach()

  message("Adding ")
  message(${targetName})
  message(${DSF_SOURCES})
  message(${libraryLists})

  add_executable(${targetName} "${DSF_SOURCES}")

  if(UNIX AND NOT ANDROID)
    set(libraryLists ${libraryLists} pthread)
  endif()

  target_link_libraries(${targetName} ${libraryLists})

  if(NOT WIN32)
      add_custom_command(TARGET ${targetName} POST_BUILD
                  COMMAND cp ${buildDir}/${targetName} ${mssBinDir}/
                  COMMAND echo "Copied ${targetName} to ${mssBinDir}/")
  endif()

endfunction()

function(build_digicert_tpm2_tool libType targetName targetDir extraDefinitions mssSrcDir buildDir mssBinDir libraryList)
    buildflags(${targetDir} ${extraDefinitions} MOCANA_FLAGS)

    buildexecutable(${libType} ${targetName} ${targetDir} ${mssSrcDir} ${buildDir} "${targetName}" ${mssBinDir} ${libraryList})

  if (WIN32)
    add_definitions("${MOCANA_FLAGS}")
  else()
    target_compile_definitions(${targetName} PRIVATE "${MOCANA_FLAGS}")
  endif()

endfunction()
